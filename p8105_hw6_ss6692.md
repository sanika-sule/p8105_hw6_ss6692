``` r
boot_sample = function(df) {
  sample_df = df |> sample_frac(size = 1, replace = TRUE)
  model = lm(tmax ~ tmin + prcp, data = sample_df)
  r_squared = broom::glance(model)[["r.squared"]]
  return(data.frame(r_squared = r_squared))
}

boot_straps =
```
